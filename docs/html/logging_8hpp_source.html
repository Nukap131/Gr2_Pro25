14inline spdlog::level::level_enum parse_level_from_env() {
15 const char* env = std::getenv("LOG_LEVEL");
16 if (!env) return spdlog::level::info;
17 std::string s(env);
18 for (auto& c : s) c = static_cast<char>(::tolower(c));
19 if (s == "trace") return spdlog::level::trace;
20 if (s == "debug") return spdlog::level::debug;
21 if (s == "info") return spdlog::level::info;
22 if (s == "warn" || s == "warning") return spdlog::level::warn;
23 if (s == "error") return spdlog::level::err;
24 if (s == "critical") return spdlog::level::critical;
25 return spdlog::level::info;
26}
