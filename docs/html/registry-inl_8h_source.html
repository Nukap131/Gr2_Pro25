57SPDLOG_INLINE void registry::initialize_logger(std::shared_ptr<logger> new_logger) {
58 std::lock_guard<std::mutex> lock(logger_map_mutex_);
59 new_logger->set_formatter(formatter_->clone());
60
61 if (err_handler_) {
62 new_logger->set_error_handler(err_handler_);
63 }
64
65 // set new level according to previously configured level or default level
66 auto it = log_levels_.find(new_logger->name());
67 auto new_level = it != log_levels_.end() ? it->second : global_log_level_;
68 new_logger->set_level(new_level);
69
70 new_logger->flush_on(flush_level_);
71
72 if (backtrace_n_messages_ > 0) {
73 new_logger->enable_backtrace(backtrace_n_messages_);
74 }
75
76 if (automatic_registration_) {
77 register_logger_(std::move(new_logger));
78 }
79}

98// set default logger.
99// default logger is stored in default_logger_ (for faster retrieval) and in the loggers_ map.
100SPDLOG_INLINE void registry::set_default_logger(std::shared_ptr<logger> new_default_logger) {
101 std::lock_guard<std::mutex> lock(logger_map_mutex_);
102 // remove previous default logger from the map
103 if (default_logger_ != nullptr) {
104 loggers_.erase(default_logger_->name());
105 }
106 if (new_default_logger != nullptr) {
107 loggers_[new_default_logger->name()] = new_default_logger;
108 }
109 default_logger_ = std::move(new_default_logger);
110}

224SPDLOG_INLINE void registry::set_levels(log_levels levels, level::level_enum *global_level) {
225 std::lock_guard<std::mutex> lock(logger_map_mutex_);
226 log_levels_ = std::move(levels);
227 auto global_level_requested = global_level != nullptr;
228 global_log_level_ = global_level_requested ? *global_level : global_log_level_;
229
230 for (auto &logger : loggers_) {
231 auto logger_entry = log_levels_.find(logger.first);
232 if (logger_entry != log_levels_.end()) {
233 logger.second->set_level(logger_entry->second);
234 } else if (global_level_requested) {
235 logger.second->set_level(*global_level);
236 }
237 }
